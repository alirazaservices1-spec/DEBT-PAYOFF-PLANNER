<script>
var selectedStrat=null;
var sampleDebts=[
  {name:'Chase Sapphire',type:'Credit Card',balance:4800,minPay:96,rate:24.99},
  {name:'Student Loan',type:'Student Loan',balance:12000,minPay:130,rate:6.8},
  {name:'Car Loan',type:'Auto Loan',balance:7200,minPay:220,rate:8.5},
  {name:'Medical Bill',type:'Medical',balance:1400,minPay:50,rate:0}
];

function calcTotalInterest(debts,order,extraMo){
  var ds=order.map(function(i){return{bal:debts[i].balance,rate:debts[i].rate/100/12,min:debts[i].minPay};});
  var totalInt=0,months=0,maxMonths=600;
  while(ds.some(function(d){return d.bal>0.01;})&&months<maxMonths){
    months++;var freed=0;
    for(var i=0;i<ds.length;i++){if(ds[i].bal<=0)continue;var interest=ds[i].bal*ds[i].rate;totalInt+=interest;ds[i].bal+=interest;}
    var extra=extraMo;
    for(var i=0;i<ds.length;i++){if(ds[i].bal<=0)continue;var pay=Math.min(ds[i].min,ds[i].bal);ds[i].bal-=pay;if(ds[i].bal<=0){freed+=ds[i].min-pay;ds[i].bal=0;}}
    var leftover=extra+freed;
    for(var i=0;i<ds.length;i++){if(ds[i].bal<=0)continue;var pay=Math.min(leftover,ds[i].bal);ds[i].bal-=pay;leftover-=pay;if(leftover<=0)break;}
  }
  return{interest:Math.round(totalInt),months:months};
}

function getOrders(){
  var indices=sampleDebts.map(function(_,i){return i;});
  return{
    avalanche:indices.slice().sort(function(a,b){return sampleDebts[b].rate-sampleDebts[a].rate;}),
    snowball:indices.slice().sort(function(a,b){return sampleDebts[a].balance-sampleDebts[b].balance;})
  };
}

function updateBars(){
  var extra=parseFloat(document.getElementById('extraPay').value)||0;
  var orders=getOrders();
  var av=calcTotalInterest(sampleDebts,orders.avalanche,extra);
  var sb=calcTotalInterest(sampleDebts,orders.snowball,extra);
  var max=Math.max(av.interest,sb.interest,1);
  document.getElementById('barAvalanche').style.setProperty('--w',Math.round(av.interest/max*85)+'%');
  document.getElementById('barSnowball').style.setProperty('--w',Math.round(sb.interest/max*85)+'%');
  document.getElementById('barCustom').style.setProperty('--w',Math.round((av.interest+sb.interest)/2/max*85)+'%');
  document.getElementById('valAvalanche').textContent='$'+av.interest.toLocaleString();
  document.getElementById('valSnowball').textContent='$'+sb.interest.toLocaleString();
  document.getElementById('valCustom').textContent='You decide';
  ['barAvalanche','barSnowball','barCustom'].forEach(function(id){
    var el=document.getElementById(id);el.style.animation='none';
    setTimeout(function(){el.style.animation='barGrow 0.9s cubic-bezier(0.34,1.56,0.64,1) both';},10);
  });
}

document.getElementById('extraPay').addEventListener('input',updateBars);
setTimeout(updateBars,600);

function selectStrat(strat){
  selectedStrat=strat;
  ['avalanche','snowball','custom'].forEach(function(s){
    document.getElementById('card-'+s).classList.toggle('selected',s===strat);
    document.getElementById('check-'+s).style.display=s===strat?'block':'none';
  });
  var btn=document.getElementById('ctaBtn');btn.classList.remove('dim');
  var names={avalanche:'Avalanche',snowball:'Snowball',custom:'Custom'};
  btn.textContent='Go with '+names[strat]+' ›';
  document.getElementById('ctaHint').textContent=strat==='avalanche'?'Mathematically optimal — saves the most money.':strat==='snowball'?'Quick wins keep you going — proven psychology.':'Your call. You can reorder anytime.';
