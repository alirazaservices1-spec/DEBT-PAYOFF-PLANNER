function tapDex() {

function logPayment(btn, name, amount) {
  btn.textContent = '✓ ' + name + ' ' + amount + ' — Logged!';
  btn.style.background = '#3A9A20';
  btn.style.color = 'white';
  btn.style.boxShadow = '0 6px 18px rgba(58,154,32,.3)';
  btn.onclick = null;
  paymentsDone++;
  burstConfetti(28);
  showXP('+50 XP 💰', '#3A9A20');
  // Auto-complete task 3 if Chase was logged
  if (name === 'Chase Sapphire') {
    setTimeout(function(){ doTask('t3', 3); }, 400);
  }
  updateDexOnProgress();
}

function doTask(id, num) {
  var el = document.getElementById(id);
  if (el.classList.contains('done')) return;
  el.classList.add('done');
  tasksDone++;
  burstConfetti(22);
  var xpMap = {1:25, 2:25, 3:50};
  showXP('+' + xpMap[num] + ' XP', '#1A6FC4');
  updateDexOnProgress();
}

    "ALL DONE! Streak is back. You are a champion! 🏆"
  ];
  if (total <= msgs.length) {
    document.getElementById('dexBubble').textContent = msgs[Math.min(total-1, msgs.length-1)];
  }
}

function restartStreak() {
  burstConfetti(70);
  setTimeout(function(){ burstConfetti(50); }, 250);
  setTimeout(function(){ burstConfetti(40); }, 500);
  showXP('🔥 STREAK RESTARTED! Day 1!', '#F5C030');
  var btn = document.querySelector('.restart-btn');
  btn.textContent = '🎉 Day 1 streak started!';
  btn.style.background = 'linear-gradient(135deg,#3A9A20,#2A8010)';
  btn.style.boxShadow = '0 10px 28px rgba(58,154,32,.4)';
  btn.onclick = null;
  document.getElementById('dexBubble').textContent = "Day 1 of your new streak! It only gets easier from here. I am so proud of you for coming back. 🏆";
  // Animate Dex
  var dex = document.getElementById('dexSvg');
  dex.style.animation = 'dexPop .6s cubic-bezier(.34,1.56,.64,1) both';
  setTimeout(function(){ dex.style.animation = ''; }, 700);
}

function burstConfetti(count) {
  var phone = document.getElementById('phone');
  var rect = phone.getBoundingClientRect();
  var cx = rect.left + rect.width / 2;
  var cy = rect.top + 160;
  var colors = ['#F5C030','#1A6FC4','#3A9A20','#F06030','#9B59B6','#E8A020','#fff'];
  for (var i = 0; i < count; i++) {
    (function(){
      var c = document.createElement('div');
      var size = 5 + Math.random()*7;
      var angle = Math.random()*360;
      var dist = 40 + Math.random()*110;
      c.style.cssText =
        'position:fixed;width:'+size+'px;height:'+size+'px;' +
        'left:'+(cx + Math.cos(angle*Math.PI/180)*dist - size/2)+'px;' +
        'top:'+(cy + Math.sin(angle*Math.PI/180)*dist - 50 - size/2)+'px;' +
        'background:'+colors[Math.floor(Math.random()*colors.length)]+';' +
        'border-radius:'+(Math.random()>.4?'50%':'2px')+';pointer-events:none;z-index:9999;' +
        'transition:transform '+(0.7+Math.random()*.8)+'s cubic-bezier(.2,.8,.3,1),' +
        'opacity '+(0.5+Math.random()*.4)+'s ease '+(0.2+Math.random()*.3)+'s;';
      document.body.appendChild(c);
      requestAnimationFrame(function(){ requestAnimationFrame(function(){
        c.style.transform = 'translate(0,0) rotate('+(Math.random()*720)+'deg)';
        c.style.opacity = '0';
      }); });
      setTimeout(function(){ if(c.parentNode) c.parentNode.removeChild(c); }, 1500);
    })();
  }
}

function showXP(text, color) {
  var phone = document.getElementById('phone');
  var rect = phone.getBoundingClientRect();
  var t = document.createElement('div');
  t.className = 'xp-toast';
  t.textContent = text;
  t.style.color = color || '#3A9A20';
  t.style.left = (rect.left + rect.width/2 - 80) + 'px';
  t.style.top = (rect.top + 130) + 'px';
  document.body.appendChild(t);
  setTimeout(function(){ if(t.parentNode) t.parentNode.removeChild(t); }, 2000);
}
