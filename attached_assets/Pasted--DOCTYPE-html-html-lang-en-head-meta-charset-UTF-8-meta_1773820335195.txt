function burstConfetti(x, y) {
  var colors = ['#F5C030','#1A6FC4','#3A9A20','#F06030','#9B59B6','#E8A020','#ffffff'];
  for (var i = 0; i < 50; i++) {
    (function(){
      var c = document.createElement('div');
      var size = 5 + Math.random() * 8;
      var angle = Math.random() * 360;
      var dist = 60 + Math.random() * 120;
      var dx = Math.cos(angle * Math.PI / 180) * dist;
      var dy = Math.sin(angle * Math.PI / 180) * dist - 40;
      c.style.cssText =
        'position:fixed;width:'+size+'px;height:'+size+'px;' +
        'left:'+(x-size/2)+'px;top:'+(y-size/2)+'px;' +
        'background:'+colors[Math.floor(Math.random()*colors.length)]+';' +
        'border-radius:'+(Math.random()>.5?'50%':'2px')+';' +
        'pointer-events:none;z-index:9999;' +
        'transition:transform '+(0.7+Math.random()*0.6)+'s cubic-bezier(.2,.8,.3,1), opacity '+(0.6+Math.random()*0.4)+'s ease '+(0.3+Math.random()*0.3)+'s;';
      document.body.appendChild(c);
      requestAnimationFrame(function(){
        requestAnimationFrame(function(){
          c.style.transform = 'translate('+dx+'px,'+dy+'px) rotate('+(Math.random()*720)+'deg)';
          c.style.opacity = '0';
        });
      });
      setTimeout(function(){ if(c.parentNode) c.parentNode.removeChild(c); }, 1400);
    })();
  }
}

function selectMethod(method, evt) {
  if (evt && evt.target.closest && evt.target.closest('.drag-list')) return;

  var wasAlreadySelected = selectedMethod === method;
  selectedMethod = method;

  document.querySelectorAll('.method-card').forEach(function(c) {
    c.classList.remove('selected', 'custom-open');
  });
  var card = document.getElementById('card-' + method);
  card.classList.add('selected');
  if (method === 'custom') card.classList.add('custom-open');

  var btn = document.getElementById('ctaBtn');
  var hint = document.getElementById('ctaHint');
  btn.classList.add('active');
  hint.style.display = 'none';

  // Celebrate every time they pick (not just first)
  if (!wasAlreadySelected) {
    setTimeout(function(){ celebrate(method); }, 50);
  }
}

// ── DRAG AND DROP ──
function initDrag() {
  var list = document.getElementById('dragList');
  if (!list) return;
  var items = list.querySelectorAll('.drag-item');
  items.forEach(function(item) {
    item.addEventListener('dragstart', function(e) {
      dragSrc = this;
      this.classList.add('dragging');
      e.dataTransfer.effectAllowed = 'move';
    });
    item.addEventListener('dragend', function() {
      this.classList.remove('dragging');
      list.querySelectorAll('.drag-item').forEach(function(i){ i.classList.remove('drag-over'); });
      updateOrder();
    });
    item.addEventListener('dragover', function(e) {
      e.preventDefault();
      e.dataTransfer.dropEffect = 'move';
      if (this !== dragSrc) {
        list.querySelectorAll('.drag-item').forEach(function(i){ i.classList.remove('drag-over'); });
        this.classList.add('drag-over');
      }
    });
    item.addEventListener('drop', function(e) {
      e.preventDefault();
      if (this !== dragSrc) {
        var allItems = Array.from(list.querySelectorAll('.drag-item'));
        var srcIdx = allItems.indexOf(dragSrc);
        var tgtIdx = allItems.indexOf(this);
        if (srcIdx < tgtIdx) list.insertBefore(dragSrc, this.nextSibling);
        else list.insertBefore(dragSrc, this);
        updateOrder();
      }
    });
  });
}

function updateOrder() {
  document.querySelectorAll('#dragList .drag-item').forEach(function(item, i) {
    var badge = item.querySelector('.drag-order');
    if (badge) badge.textContent = i + 1;
  });
}

document.addEventListener('DOMContentLoaded', function(){ initDrag(); });
