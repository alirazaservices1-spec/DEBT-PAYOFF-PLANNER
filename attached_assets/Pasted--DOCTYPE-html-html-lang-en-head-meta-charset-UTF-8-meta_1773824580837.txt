// ── COMMIT SCREEN ──
function toggleCommit() {
  committed = !committed;
  var box = document.getElementById('commitBox');
  var check = document.getElementById('cbCheck');
  var btn = document.getElementById('commitBtn');
  box.classList.toggle('checked', committed);
  btn.classList.toggle('ready', committed);
}

function doCommit() {
  if (!committed) return;
  var phone = document.getElementById('phone1');
  burstConfetti(phone, 60);
  setTimeout(function(){ burstConfetti(phone, 40); }, 300);
  showXP(phone, '🤝 COMMITTED! +200 XP');
  // Animate Dex jump
  var dex = document.getElementById('commitDex');
  dex.style.animation = 'none';
  void dex.offsetWidth;
  dex.style.animation = 'dexJump .7s cubic-bezier(.34,1.56,.64,1) both';
  setTimeout(function(){ dex.style.animation = 'coachIdle 2.8s ease-in-out infinite'; }, 800);
}

// ── CONFETTI ──
function burstConfetti(phone, count) {
  var rect = phone.getBoundingClientRect();
  var cx = rect.left + rect.width / 2;
  var cy = rect.top + 120;
  var colors = ['#F5C030','#1A6FC4','#3A9A20','#F06030','#9B59B6','#E8A020','#fff'];
  for (var i = 0; i < count; i++) {
    (function(){
      var c = document.createElement('div');
      var size = 5 + Math.random()*8;
      var angle = Math.random()*360;
      var dist = 40 + Math.random()*120;
      var dx = Math.cos(angle*Math.PI/180)*dist;
      var dy = Math.sin(angle*Math.PI/180)*dist - 40;
      c.style.cssText =
        'position:fixed;width:'+size+'px;height:'+size+'px;' +
        'left:'+(cx-size/2)+'px;top:'+(cy-size/2)+'px;' +
        'background:'+colors[Math.floor(Math.random()*colors.length)]+';' +
        'border-radius:'+(Math.random()>.4?'50%':'2px')+';' +
        'pointer-events:none;z-index:9999;' +
        'transition:transform '+(0.7+Math.random()*.8)+'s cubic-bezier(.2,.8,.3,1),' +
        'opacity '+(0.5+Math.random()*.5)+'s ease '+(0.25+Math.random()*.3)+'s;';
      document.body.appendChild(c);
      requestAnimationFrame(function(){
        requestAnimationFrame(function(){
          c.style.transform = 'translate('+dx+'px,'+dy+'px) rotate('+(Math.random()*720)+'deg)';
          c.style.opacity = '0';
        });
      });
      setTimeout(function(){ if(c.parentNode) c.parentNode.removeChild(c); }, 1500);
    })();
  }
}

function showXP(phone, text) {
  var rect = phone.getBoundingClientRect();
  var t = document.createElement('div');
  t.className = 'xp-toast';
  t.textContent = text;
  t.style.left = (rect.left + rect.width/2 - 90) + 'px';
  t.style.top = (rect.top + 80) + 'px';
  document.body.appendChild(t);
  setTimeout(function(){ if(t.parentNode) t.parentNode.removeChild(t); }, 2000);
}
