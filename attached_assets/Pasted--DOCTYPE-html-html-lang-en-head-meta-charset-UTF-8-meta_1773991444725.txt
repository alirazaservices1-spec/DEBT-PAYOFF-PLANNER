<script>
let debts=[
  {id:1,name:"Chase Sapphire",type:"Credit Card",balance:12400,apr:22.99,payment:248,icon:"💳",progress:15},
  {id:2,name:"Sallie Mae",type:"Student Loan",balance:8750,apr:6.8,payment:180,icon:"🎓",progress:40},
  {id:3,name:"Toyota Financial",type:"Auto Loan",balance:5200,apr:7.9,payment:134,icon:"🚗",progress:62},
  {id:4,name:"Chase Freedom",type:"Credit Card",balance:2000,apr:19.99,payment:50,icon:"💳",progress:78},
];
let sortAsc=false;
let editIdx=null;
let deleteIdx=null;
let openCard=null;

function aprClass(apr){return apr>15?'apr-high':apr>10?'apr-mid':'apr-low';}

function render(){
  const list=document.getElementById('debtList');
  list.innerHTML='';
  debts.forEach((d,i)=>{
    const isOpen=openCard===i;
    const div=document.createElement('div');
    div.className='df-debt-card';
    div.innerHTML=`
      <div class="df-progress-bar"><div class="df-progress-fill" style="width:${d.progress}%"></div></div>
      <div class="df-debt-main" onclick="toggleCard(${i})">
        <div class="df-debt-icon" style="background:#f4efe5;font-size:20px">${d.icon}</div>
        <div class="df-debt-info">
          <div class="df-debt-name">${d.name}</div>
          <div class="df-debt-sub">${d.type} · $${d.payment}/mo min</div>
        </div>
        <div class="df-debt-right">
          <div class="df-debt-balance">$${d.balance.toLocaleString()}</div>
          <div class="df-debt-apr ${aprClass(d.apr)}">${d.apr}% APR</div>
        </div>
        <span class="df-chevron${isOpen?' open':''}">›</span>
      </div>
      <div class="df-debt-actions${isOpen?' show':''}">
        <button class="df-action-btn btn-edit" onclick="openEdit(${i},event)">✏️ Edit</button>
        <button class="df-action-btn btn-adjust" onclick="openAdjust(${i},event)">📊 Adjust</button>
        <button class="df-action-btn btn-delete" onclick="openDelete(${i},event)">🗑️ Delete</button>
      </div>`;
    list.appendChild(div);
  });
  updateTotals();
}

function updateTotals(){
  const total=debts.reduce((s,d)=>s+d.balance,0);
  document.getElementById('totalAmount').textContent='$'+total.toLocaleString();
  document.getElementById('debtCount').textContent=debts.length+' debt'+(debts.length===1?'':'s');
  document.getElementById('sectionTitle').textContent=debts.length+' account'+(debts.length===1?'':'s');
}

function toggleCard(i){openCard=openCard===i?null:i;render();}
function toggleSort(){sortAsc=!sortAsc;debts.sort((a,b)=>sortAsc?a.balance-b.balance:b.balance-a.balance);render();}

function openAdd(){
  editIdx=null;
  document.getElementById('modalTitle').textContent='Add Debt';
  document.getElementById('f-name').value='';
  document.getElementById('f-type').value='Credit Card';
  document.getElementById('f-balance').value='';
  document.getElementById('f-apr').value='';
  document.getElementById('f-payment').value='';
  document.getElementById('editModal').classList.add('show');
}
function openEdit(i,e){
  e.stopPropagation();editIdx=i;
  const d=debts[i];
  document.getElementById('modalTitle').textContent='Edit '+d.name;
  document.getElementById('f-name').value=d.name;
  document.getElementById('f-type').value=d.type;
  document.getElementById('f-balance').value=d.balance;
  document.getElementById('f-apr').value=d.apr;
  document.getElementById('f-payment').value=d.payment;
  document.getElementById('editModal').classList.add('show');
}
function openAdjust(i,e){
  e.stopPropagation();editIdx=i;
  const d=debts[i];
  document.getElementById('modalTitle').textContent='Adjust Payment — '+d.name;
  document.getElementById('f-name').value=d.name;
  document.getElementById('f-type').value=d.type;
  document.getElementById('f-balance').value=d.balance;
  document.getElementById('f-apr').value=d.apr;
  document.getElementById('f-payment').value=d.payment;
  document.getElementById('editModal').classList.add('show');
}
function closeModal(){document.getElementById('editModal').classList.remove('show');}
function saveDebt(){
  const nm=document.getElementById('f-name').value.trim();
  if(!nm)return;
  const type=document.getElementById('f-type').value;
  const obj={
    id:Date.now(),name:nm,type:type,
    balance:parseFloat(document.getElementById('f-balance').value)||0,
    apr:parseFloat(document.getElementById('f-apr').value)||0,
    payment:parseFloat(document.getElementById('f-payment').value)||0,
    icon:type==='Credit Card'?'💳':type==='Student Loan'?'🎓':type==='Auto Loan'?'🚗':type==='Medical'?'🏥':'📄',
    progress:5
  };
  if(editIdx!==null)debts[editIdx]=obj;
  else debts.push(obj);
  closeModal();openCard=null;render();
}
function openDelete(i,e){
  e.stopPropagation();deleteIdx=i;
  document.getElementById('deleteSub').textContent='Remove '+debts[i].name+' from your payoff plan?';
  document.getElementById('deleteModal').classList.add('show');
}
function closeDeleteModal(){document.getElementById('deleteModal').classList.remove('show');}
function confirmDelete(){debts.splice(deleteIdx,1);closeDeleteModal();openCard=null;render();}

render();
</script>
