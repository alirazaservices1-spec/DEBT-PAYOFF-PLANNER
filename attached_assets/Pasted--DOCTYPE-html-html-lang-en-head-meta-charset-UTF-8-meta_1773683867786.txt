function renderCal(){
  document.getElementById('calMonthLabel').textContent=MONTHS_FULL[calMonth]+' '+calYear;
  var grid=document.getElementById('calGrid'); grid.innerHTML='';
  DAYS.forEach(function(d){var el=document.createElement('div');el.className='cal-day-label';el.textContent=d;grid.appendChild(el);});
  var first=new Date(calYear,calMonth,1).getDay(), daysIn=new Date(calYear,calMonth+1,0).getDate();
  var today=new Date();
  for(var i=0;i<first;i++){var el=document.createElement('div');el.className='cal-day empty';grid.appendChild(el);}
  for(var d=1;d<=daysIn;d++){(function(day){var el=document.createElement('div');el.className='cal-day';el.textContent=day;
    var isToday=day===today.getDate()&&calMonth===today.getMonth()&&calYear===today.getFullYear();
    var isSel=selDate&&day===selDate.d&&calMonth===selDate.m&&calYear===selDate.y;
    if(isSel)el.className+=' selected';else if(isToday)el.className+=' today';
    el.onclick=function(){selDate={d:day,m:calMonth,y:calYear};var disp=document.getElementById('calDisplay');disp.textContent=MONTHS_SHORT[calMonth]+' '+day+', '+calYear;disp.className='';calOpen=false;document.getElementById('calPopup').style.display='none';};
    grid.appendChild(el);})(d);}
}

function showError(msg){var err=document.getElementById('formError');err.textContent=msg;err.style.display='block';err.style.animation='none';setTimeout(function(){err.style.animation='shake 0.35s ease';},10);}
function monthsToPayoff(balance,apr,minPay){if(minPay<=0||balance<=0)return null;var monthly=apr/100/12;if(monthly===0)return Math.ceil(balance/minPay);if(minPay<=balance*monthly)return 999;return Math.ceil(-Math.log(1-(balance*monthly)/minPay)/Math.log(1+monthly));}
function fmtMonths(n){if(!n||n>=999)return '30y+';if(n<=12)return n+'mo';var y=Math.floor(n/12),m=n%12;return y+'y'+(m?' '+m+'mo':'');}
function fmtCurrency(v){return '$'+v.toLocaleString('en-US',{minimumFractionDigits:0,maximumFractionDigits:0});}
function fmtRate(d){return d.isVar?(d.rateMin.toFixed(2)+'%–'+d.rateMax.toFixed(2)+'%'):d.rate.toFixed(2)+'%';}
function esc(s){return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');}

function addDebt(){
  document.getElementById('formError').style.display='none';
  var name=document.getElementById('debtName').value.trim()||selType;
  var balance=parseFloat(document.getElementById('balance').value);
  var minPay=parseFloat(document.getElementById('minPay').value);
  var rate=parseFloat(document.getElementById('rate').value);
  if(isNaN(balance)||balance<=0){showError('Please enter a valid balance.');return;}
  if(isNaN(minPay)||minPay<0){showError('Please enter a minimum payment.');return;}
  if(isNaN(rate)||rate<0){showError('Please enter an interest rate (0 if none).');return;}
  var debt={id:Date.now(),name:name,type:selType,balance:balance,minPay:minPay,rate:rate,isVar:isVar,
    rateMin:isVar?parseFloat(document.getElementById('rateMin').value)||rate:null,
    rateMax:isVar?parseFloat(document.getElementById('rateMax').value)||rate:null,
    dueDate:selDate?MONTHS_SHORT[selDate.m]+' '+selDate.d:null};
  debts.push(debt);
  resetForm();
  showCelebration(debt);
}

}

function showCelebration(debt){
  launchConfetti();
  var idx=Math.min(debts.length-1,MESSAGES.length-1);
  var msg=MESSAGES[idx];
  var variantFn=DEX_VARIANTS[(debts.length-1)%DEX_VARIANTS.length];
  document.getElementById('celBadge').textContent='Debt #'+debts.length+' Added!';
  document.getElementById('celHeadline').innerHTML=msg.headline;
  document.getElementById('celSub').textContent=msg.sub;
  document.getElementById('celDex').innerHTML='<div class="pulse-ring"></div>'+variantFn();
  var mo=monthsToPayoff(debt.balance,debt.rate,debt.minPay);
  document.getElementById('celCard').innerHTML=
    '<div class="cel-card-name">'+esc(debt.name)+' <span class="cel-tag">'+esc(debt.type)+'</span></div>'+
    '<div class="cel-stats">'+
      '<div class="cel-stat"><span class="cel-stat-val">'+fmtCurrency(debt.balance)+'</span><span class="cel-stat-lbl">Balance</span></div>'+
      '<div class="cel-stat"><span class="cel-stat-val">'+(debt.isVar?'<span class="var-badge-sm">var</span> ':'')+fmtRate(debt)+'</span><span class="cel-stat-lbl">APR</span></div>'+
      '<div class="cel-stat"><span class="cel-stat-val">'+fmtCurrency(debt.minPay)+'</span><span class="cel-stat-lbl">Min/mo</span></div>'+
      '<div class="cel-stat"><span class="cel-stat-val">'+(mo?fmtMonths(mo):'—')+'</span><span class="cel-stat-lbl">Payoff est.</span></div>'+
    '</div>';
  var totalBal=debts.reduce(function(s,d){return s+d.balance;},0);
  var totalMin=debts.reduce(function(s,d){return s+d.minPay;},0);
  var avgRate=debts.reduce(function(s,d){return s+d.rate*d.balance;},0)/(totalBal||1);
  var barPct=Math.min(100,debts.length*14);
  document.getElementById('celProgress').innerHTML=
    '<div class="prog-label">Running total — '+debts.length+' debt'+(debts.length>1?'s':'')+' tackled</div>'+
    '<div class="prog-bar-wrap"><div class="prog-bar" style="width:'+barPct+'%"></div></div>'+
    '<div class="prog-stats">'+
      '<span class="prog-stat"><strong>'+fmtCurrency(totalBal)+'</strong> total</span>'+
      '<span class="prog-stat"><strong>'+avgRate.toFixed(1)+'%</strong> avg APR</span>'+
      '<span class="prog-stat"><strong>'+fmtCurrency(totalMin)+'</strong>/mo</span>'+
    '</div>';
  document.getElementById('celBtns').innerHTML=
    '<button class="btn-secondary" onclick="goBack()">+ Add Another Debt</button>'+

  document.getElementById('celScreen').classList.remove('hidden');

}
