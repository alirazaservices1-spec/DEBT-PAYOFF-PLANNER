// ── REAL DEBT PAYOFF CALC: simulate month-by-month ──
// debtData represents the debts from step 1 (demo values summing to $26,400 at avg 22% APR)
var debtData = [
  { balance: 12400, apr: 24.99, minPayment: 248 },
  { balance: 8200,  apr: 22.49, minPayment: 164 },
  { balance: 5800,  apr: 18.00, minPayment: 116 }
];

function calcDebtPayoff(extraMonthly) {
  // Simulate month-by-month payoff with avalanche method + extra payment
  // Returns months to pay off all debts
  var balances = debtData.map(function(d){ return d.balance; });
  var rates    = debtData.map(function(d){ return d.apr / 100 / 12; });
  var mins     = debtData.map(function(d){ return d.minPayment; });
  var totalMin = mins.reduce(function(s,m){ return s+m; }, 0);
  var monthlyPayment = totalMin + extraMonthly;

  var months = 0;
  var maxMonths = 600;
  while (months < maxMonths) {
    // Apply interest to all remaining balances
    var allPaid = true;
    for (var i = 0; i < balances.length; i++) {
      if (balances[i] > 0) {
        balances[i] += balances[i] * rates[i];
        allPaid = false;
      }
    }
    if (allPaid) break;
    months++;

    // Pay minimums on all, then attack highest-rate first with remainder
    var remaining = monthlyPayment;
    // Pay minimums first
    for (var i = 0; i < balances.length; i++) {
      if (balances[i] > 0) {
        var pay = Math.min(mins[i], balances[i]);
        balances[i] -= pay;
        remaining -= pay;
      }
    }
    // Apply extra to highest balance (avalanche: already sorted by apr desc in demo data)
    for (var i = 0; i < balances.length; i++) {
      if (balances[i] > 0 && remaining > 0) {
        var pay = Math.min(remaining, balances[i]);
        balances[i] -= pay;
        remaining -= pay;
      }
    }
    // Round tiny balances to zero
    for (var i = 0; i < balances.length; i++) {
      if (balances[i] < 0.01) balances[i] = 0;
    }
    if (balances.every(function(b){ return b===0; })) break;
  }
  return months;
}

function calcDreamMonths(savingsMonthly, goal) {
  // Compound savings at 4% APR
  var monthlyRate = 0.04 / 12;
  if (savingsMonthly <= 0 || goal <= 0) return 999;
  if (monthlyRate === 0) return Math.ceil(goal / savingsMonthly);
  return Math.ceil(Math.log(1 + goal * monthlyRate / savingsMonthly) / Math.log(1 + monthlyRate));
}

function fmt(months) {
  if (months >= 999) return '---';
  if (months <= 1) return '1 mo.';
  if (months < 24) return months + ' mo.';
  var y = Math.floor(months / 12);
  var m = months % 12;
  return y + ' yr' + (y>1?'s':'') + (m>0 ? ' ' + m + ' mo.' : '');
}

function toDate(months) {
  var d = new Date();
  d.setMonth(d.getMonth() + months);
  return d.toLocaleDateString('en-US', { month: 'short', year: 'numeric' });
}

function updateVision() {
  var extraMonthly = Math.round(extraPerDay * 30.44);
  var debtMonths = calcDebtPayoff(extraMonthly);
  // After debt paid: redirect full old payment + extra into savings
  var totalMin = debtData.reduce(function(s,d){ return s+d.minPayment; }, 0);
  var savingsMonthly = totalMin + extraMonthly;

  // Always update debt-free stat even without a dream selected
  var dfEl = document.getElementById('vsDebtFree');
  if (dfEl) dfEl.textContent = fmt(debtMonths);
  var dfDateEl = document.getElementById('vsDebtDate');
  if (dfDateEl) dfDateEl.textContent = toDate(debtMonths);

  if (!dreamName || dreamCost <= 0) {
    // Show debt-free date even without a dream
    var vc = document.getElementById('visionCard');
    if (dreamCost <= 0 && dreamName) {
      // Dream selected but no cost yet - show partial
      vc.classList.add('show');
      document.getElementById('visionHeadline').textContent = dreamName + ': enter a cost above to see your timeline 👆';
      document.getElementById('vsDebtFree').textContent = fmt(debtMonths);
      document.getElementById('vsDreamReady').textContent = '...';
    } else {
      vc.classList.remove('show');
    }
    updateSliderImpact(debtMonths, null, savingsMonthly, extraMonthly);
    return;
  }

  var dreamMonths = calcDreamMonths(savingsMonthly, dreamCost);
  var totalMonths = debtMonths + dreamMonths;

  var headline;
  if (dreamMonths <= 6) headline = dreamName + ' in just ' + fmt(dreamMonths) + ' after your last payment. This could be very soon! 🚀';
  else if (dreamMonths <= 24) headline = 'Just ' + fmt(dreamMonths) + ' of saving after debt-free and ' + dreamName + ' is yours. 🌟';
  else headline = dreamName + ' in ' + fmt(dreamMonths) + ' after your debt is gone. Worth every step. 💪';

  // Big bold headline: "[Dream] in X yrs Y mo. 💪"
  var totalDisplay = fmt(totalMonths);
  document.getElementById('visionHeadline').textContent = dreamName + ' in ' + totalDisplay + ' 💪';

  // Sub lines
  document.getElementById('vsDebtFree').textContent = fmt(debtMonths);
  document.getElementById('vsDreamReady').textContent = fmt(dreamMonths);
  var dn = document.getElementById('vsDreamName');
  if (dn) dn.textContent = dreamName || 'your dream';
  var sv = document.getElementById('vsSavings');
  if (sv) sv.textContent = 'Saving $' + Math.round(savingsMonthly).toLocaleString() + '/mo at 4% APY after debt';
  document.getElementById('vsTotal').textContent = totalDisplay;

  document.getElementById('visionCard').classList.add('show');
  updateSliderImpact(debtMonths, dreamMonths, savingsMonthly, extraMonthly);

  if (!window._visionShown) {
    window._visionShown = true;
    setTimeout(celebrateDream, 200);
  }
}

function updateSliderImpact(debtMonths, dreamMonths, savingsMonthly, extraMonthly) {
  var el = document.getElementById('sliderImpact');
  if (!el) return;
  var baseDebt = calcDebtPayoff(0);
  var saved = baseDebt - debtMonths;
  if (extraMonthly === 0) {
    el.textContent = 'Even $0.50/day = $15/mo less interest and a sooner dream!';
  } else {
    el.textContent = '+$' + extraMonthly + '/mo saves ' + saved + ' month' + (saved!==1?'s':'') + ' of debt' + (dreamName ? ' and gets your ' + dreamName + ' sooner.' : '.');
  }
}
