function launchConfetti(){
  var wrap=document.getElementById('confWrap');wrap.innerHTML='';
  var colors=['#C07820','#E8960A','#EAA835','#7A3A0C','#D09828','#F5C842','#A86010','#639922'];
  for(var i=0;i<36;i++){(function(){
    var el=document.createElement('div');el.className='conf';
    el.style.left=(4+Math.random()*92)+'%';el.style.top='-10px';
    el.style.background=colors[Math.floor(Math.random()*colors.length)];
    el.style.width=(5+Math.random()*8)+'px';el.style.height=(5+Math.random()*8)+'px';
    el.style.borderRadius=Math.random()>0.45?'50%':'2px';
    el.style.animationDelay=(Math.random()*0.9)+'s';
    el.style.animationDuration=(1.2+Math.random()*0.8)+'s';
    wrap.appendChild(el);
    setTimeout(function(){if(el.parentNode)el.parentNode.removeChild(el);},2800);
  })();}
}

function playFanfare(){
  try{
    var ac=new(window.AudioContext||window.webkitAudioContext)();
    [[523,0],[659,0.08],[784,0.16],[1047,0.26],[1319,0.38]].forEach(function(n){
      var o=ac.createOscillator(),g=ac.createGain();
      o.connect(g);g.connect(ac.destination);
      o.frequency.value=n[0];o.type='sine';
      g.gain.setValueAtTime(0.1,ac.currentTime+n[1]);
      g.gain.exponentialRampToValueAtTime(0.001,ac.currentTime+n[1]+0.45);
      o.start(ac.currentTime+n[1]);o.stop(ac.currentTime+n[1]+0.5);
    });
  }catch(e){}
}

function goLaunch(){
  launchConfetti();
  playFanfare();
  var prefs=[];
  if(toggleState.togEmail) prefs.push({icon:'📧',lbl:'Email coaching enabled'});
  if(toggleState.togNotif) prefs.push({icon:'🔔',lbl:'App notifications enabled'});
  if(toggleState.togSound) prefs.push({icon:'🔊',lbl:'Sound alerts enabled'});
  if(prefs.length===0) prefs.push({icon:'👤',lbl:'Solo mode — you\'ve got this!'});
  var html='';
  prefs.forEach(function(p){
    html+='<div class="pref-row"><span class="pref-icon">'+p.icon+'</span><span class="pref-lbl">'+p.lbl+'</span>'+
      '<div class="pref-check"><svg width="11" height="9" viewBox="0 0 11 9"><path d="M1 4.5L4 7.5L10 1" stroke="white" stroke-width="1.8" fill="none" stroke-linecap="round" stroke-linejoin="round"/></svg></div></div>';
  });
  document.getElementById('launchPrefs').innerHTML=html;
  setTimeout(function(){
    document.getElementById('coachScreen').classList.add('hidden');
    document.getElementById('launchScreen').classList.remove('hidden');
    setTimeout(function(){launchConfetti();},400);
  },320);
}

function goApp(){
  launchConfetti();
  alert('Opening Day 1 Dashboard! Email: '+toggleState.togEmail+' | Notifications: '+toggleState.togNotif+' | Sound: '+toggleState.togSound);
}
