<script>
  // Respect system preference on load
  var dark = window.matchMedia('(prefers-color-scheme: dark)').matches;
  updateToggleLabel();

  function toggleMode() {
    dark = !dark;
    if (dark) {
      document.documentElement.style.setProperty('color-scheme', 'dark');
      // manually override vars for forced toggle
      applyVars({
        '--bg-page':'#1A1410','--bg-phone':'#2A2018','--phone-border':'#0A0400',
        '--blob1':'rgba(200,128,30,0.12)','--blob2':'rgba(200,128,30,0.08)',
        '--notch':'rgba(255,220,160,0.12)','--eyebrow':'#C09050',
        '--title':'#F0E8D0','--accent':'#E8A030','--sub':'#B09070',
        '--btn-bg':'#D08828','--btn-text':'#FEF8EE','--btn-hover':'#B87020',
        '--trust':'#A08060','--lock-fill':'#A08060','--lock-hole':'#2A2018',
        '--shadow':'rgba(0,0,0,0.6), 0 6px 24px rgba(0,0,0,0.4)'
      });
    } else {
      document.documentElement.style.setProperty('color-scheme', 'light');
      applyVars({
        '--bg-page':'#F5F0E8','--bg-phone':'#EDE8D5','--phone-border':'#180600',
        '--blob1':'rgba(200,128,30,0.09)','--blob2':'rgba(200,128,30,0.07)',
        '--notch':'rgba(24,6,0,0.16)','--eyebrow':'#9A7240',
        '--title':'#2A1808','--accent':'#C07820','--sub':'#6B4E30',
        '--btn-bg':'#C07820','--btn-text':'#FEF8EE','--btn-hover':'#A86510',
        '--trust':'#9A7240','--lock-fill':'#9A7240','--lock-hole':'#EDE8D5',
        '--shadow':'rgba(0,0,0,0.32), 0 6px 16px rgba(0,0,0,0.12)'
      });
    }
    updateToggleLabel();
  }

  function applyVars(vars) {
    Object.keys(vars).forEach(function(k) {
      document.documentElement.style.setProperty(k, vars[k]);
    });
  }

  function updateToggleLabel() {
    document.getElementById('modeBtn').textContent = dark ? '☀️ Light' : '🌙 Dark';
  }

  function handleStart() {
    var btn = document.getElementById('startBtn');
    btn.textContent = "Let's go! 🎉";
    btn.style.background = '#3B6D11';
    setTimeout(function() {
      btn.textContent = 'Get Started →';
      btn.style.background = '';
    }, 1500);
  }
</script>
